35      * Runs the widget.
36      */
37     public function run()
38     {
39         // todo: think of a better way of doing this.
40         $script = "jQuery('{$this->selector}').attr('data-spy', 'scroll');";
41         if (isset($this->target))
42             $script .= "jQuery('{$this->selector}').attr('data-target', '{$this->target}');";
43         if (isset($this->offset))
44             $script .= "jQuery('{$this->selector}').attr('data-offset', '{$this->offset}');";
45         Yii::app()->clientScript->registerScript($this->getId(), $script, CClientScript::POS_BEGIN);
46         $this->registerEvents($this->selector, $this->events);
47     }
