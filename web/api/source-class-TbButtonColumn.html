34     /**
35      * Initializes the default buttons (view, update and delete).
36      */
37     protected function initDefaultButtons()

44             $this->buttons['update']['icon'] = $this->updateButtonIcon;

49     /**
50      * Renders a link button.
51      * @param string $id the ID of the button
52      * @param array $button the button configuration which may contain 'label', 'url', 'imageUrl' and 'options' elements.
53      * @param integer $row the row number (zero-based)
54      * @param mixed $data the data object associated with the row
55      */
56     protected function renderButton($id, $button, $row, $data)
57     {
58         if (isset($button['visible']) && !$this->evaluateExpression($button['visible'], array('row' => $row, 'data' => $data)))
59             return;
60 
61         $url = TbHtml::popOption('url', $button, '#');
62         if(strcmp($url, '#') !== 0)
63             $url = $this->evaluateExpression($url, array('data' => $data, 'row' => $row));
64 
65         $imageUrl = TbHtml::popOption('imageUrl', $button, false);
66         $label = TbHtml::popOption('label', $button, $id);
67         $options = TbHtml::popOption('options', $button, array());
68 
69         $options = TbHtml::defaultOption('title', $label, $options);
70         $options = TbHtml::defaultOption('rel', 'tooltip', $options);
71 
72         if ($icon = TbHtml::popOption('icon', $button, false ))
73         {
74             /* todo: not sure if we require this anymore */
75             if (strpos($icon, 'icon') === false)
76                 $icon = 'icon-' . implode(' icon-', explode(' ', $icon));
77 
78             echo CHtml::link('<i class="' . $icon. '"></i>', $url, $options);
79 
80         } else if ($imageUrl && is_string($imageUrl))
81             echo CHtml::link(CHtml::image($imageUrl, $label), $url, $options);
82         else
83             echo CHtml::link($label, $url, $options);
84     }
85 }
